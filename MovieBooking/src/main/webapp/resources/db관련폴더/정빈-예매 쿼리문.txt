SELECT
	MOVIE.mov_code as movCode,
	MOVIE.mov_korName as movKorName,
	MOVIE.mov_grade as movGrade,
	ROUND(GRADE.gra_acGrade/GRADE.gra_totalNum) as grade ,
	ROUND(GRADE.gra_acGrade/GRADE.gra_totalNum/2) as gradestar
	
	
FROM
	MOVIE JOIN ANALYSIS ON ANALYSIS.mov_code = MOVIE.mov_code
			JOIN GRADE ON GRADE.mov_code = MOVIE.mov_code
	
ORDER BY 
	ANALYSIS.anl_totalTicketCount DESC
